56  {
57  // Setup
58  SD_JsonSerializer serializer = new SD_JsonSerializer();
59  SD_Encoder encoder = new SD_Encoder();
60  encoding ??= Encoding.UTF8;
61  objectToSave ??= default(T);
62  if (string.IsNullOrEmpty(filename))
63  throw new System.ArgumentNullException(nameof(filename));
64  string filePath = GetFilePath(filename);
65 
66  // Build IO stream
67  Stream stream;
68  #if !UNITY_SAMSUNGTV && !UNITY_TVOS && !UNITY_WEBGL
69  #if UNITY_WSA || UNITY_WINRT

71  #else

73  #endif
74  #endif
75  if (!encryptionPassword.IsNullEmptyOrWhiteSpace())
76  {
77  stream = new MemoryStream();
78  }
79  else
80  {
81  #if !UNITY_SAMSUNGTV && !UNITY_TVOS && !UNITY_WEBGL
82  if (Utils.IsIOSupported())
83  {
84  #if UNITY_WSA || UNITY_WINRT
85  stream = new MemoryStream ();
86  #else
87  stream = File.Create(filePath);
88  #endif
89  }
90  else
91  {
92  stream = new MemoryStream();
93  }
94  #else
95  stream = new MemoryStream ();
96  #endif
97  }
98 
99  // Serialize object
100  serializer.Serialize(objectToSave, stream, encoding);
101  if (!encryptionPassword.IsNullEmptyOrWhiteSpace())
102  {
103  string data = System.Convert.ToBase64String(((MemoryStream) stream).ToArray());
104  string encoded = encoder.Encode(data, encryptionPassword);
105  #if !UNITY_SAMSUNGTV && !UNITY_TVOS && !UNITY_WEBGL
106  if (Utils.IsIOSupported())
107  {

110  #else

112  #endif
113  }
114  else
115  {
116  PlayerPrefs.SetString(filePath, encoded);
117  PlayerPrefs.Save();
118  }
119  #else
120  PlayerPrefs.SetString ( filePath, encoded );
121  PlayerPrefs.Save ();
122  #endif
123  }
124  else if (!Utils.IsIOSupported())
125  {
126  string data = encoding.GetString(((MemoryStream) stream).ToArray());
