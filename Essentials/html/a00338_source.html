38  public static Rect GetWorldRect(this RectTransform rt)
39  {
40  // Be aware: a standard Rect has the position as the upper-left corner,
41  // and I think the Unity UI stuff somehow repurposes this to instead point to the
42  // lower-left. I'm not 100% sure on this, but I've had some unexplained wierdnesses.
43  Vector3[] corners = new Vector3[4];
44  rt.GetWorldCorners(corners);
45  Vector2 size = new Vector2(corners[2].x - corners[1].x, corners[1].y - corners[0].y);
46  return new Rect(new Vector2(corners[1].x, -corners[1].y), size);
47  }

178  // Set[edgeName](float) is similar to setting the "Left" etc variables in the inspector. Unlike the inspector, these
179  // can be used regardless of anchor position. Be warned, there's a reason the functionality
180  // is hidden in the editor, as the behavior is unintuitive when adjusting the parent's rect.
181  // If you're calling these every frame or otherwise updating frequently, shouldn't be a problem, though.
182  //
183  //Keep in mind that these functions all use standard directions when determining positioning; this means
184  // that unlike the inspector, positive ALWAYS means to the right/top, and negative ALWAYS means to the left/
185  // bottom. If you want true inspector functionality, use Left() and so on, below.
186  //
187  //E.g., SetLeftEdge(-10) will turn
188  /*
189  .__________.
190  | |
191  | |
192  | [ ] |
193  | |
194  |__________|
195 
196  into
197  .__________.
198  | |
199  | |
200  [ ] |
201  | |
202  |__________|
203 
204  [ ] is the RectTransform, the bigger square is the parent
205  */
210  public static void SetLeft(this RectTransform rt, float distance)
211  {
212  float xmin = rt.GetParent().rect.xMin;
213  float anchorFactor = rt.anchorMin.x * 2 - 1;
214 
215  rt.offsetMin = new Vector2(xmin + (xmin * anchorFactor) + distance, rt.offsetMin.y);
216  }

222  {
223  float xmax = rt.GetParent().rect.xMax;
224  float anchorFactor = rt.anchorMax.x * 2 - 1;
225 
226  rt.offsetMax = new Vector2(xmax - (xmax * anchorFactor) + distance, rt.offsetMax.y);
227  }
