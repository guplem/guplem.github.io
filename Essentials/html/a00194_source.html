548  private fsResult InternalSerialize_1_ProcessCycles(Type storageType, Type overrideConverterType, object instance, out fsData data) {
549  // We have an object definition to serialize.
550  try {
551  // Note that we enter the reference group at the beginning of serialization so that we support
552  // references that are at equal serialization levels, not just nested serialization levels, within
553  // the given subobject. A prime example is serialization a list of references.
554  _references.Enter();
555 
556  // This type does not need cycle support.
557  var converter = GetConverter(instance.GetType(), overrideConverterType);
558  if (converter.RequestCycleSupport(instance.GetType()) == false) {
559  return InternalSerialize_2_Inheritance(storageType, overrideConverterType, instance, out data);
560  }
561 
562  // We've already serialized this object instance (or it is pending higher up on the call stack).
563  // Just serialize a reference to it to escape the cycle.
564  //
565  // note: We serialize the int as a string to so that we don't lose any information
566  // in a conversion to/from double.
567  if (_references.IsReference(instance)) {
568  data = fsData.CreateDictionary();
569  _lazyReferenceWriter.WriteReference(_references.GetReferenceId(instance), data.AsDictionary);
570  return fsResult.Success;
571  }
572 
573  // Mark inside the object graph that we've serialized the instance. We do this *before*
574  // serialization so that if we get back into this function recursively, it'll already
575  // be marked and we can handle the cycle properly without going into an infinite loop.
576  _references.MarkSerialized(instance);
577 
578  // We've created the cycle metadata, so we can now serialize the actual object.
579  // InternalSerialize will handle inheritance correctly for us.
580  var result = InternalSerialize_2_Inheritance(storageType, overrideConverterType, instance, out data);
581  if (result.Failed) return result;
582 
583  _lazyReferenceWriter.WriteDefinition(_references.GetReferenceId(instance), data);
584 
585  return result;
586  }
587  finally {
588  if (_references.Exit()) {
589  _lazyReferenceWriter.Clear();
590  }
591  }

685  // We handle object references first because we could be deserializing a cyclic type that is
686  // inherited. If that is the case, then if we handle references after inheritances we will try
687  // to create an object instance for an abstract/interface type.
688 
689  // While object construction should technically be two-pass, we can do it in
690  // one pass because of how serialization happens. We traverse the serialization
691  // graph in the same order during serialization and deserialization, so the first
692  // time we encounter an object it'll always be the definition. Any times after that

694  // will have *always* already encountered the definition for it.

697  result = _references.GetReferenceObject(refId);

699  return fsResult.Success;

754 
755  // If the serialized state contains type information, then we need to make sure to update our
756  // objectType and data to the proper values so that when we construct an object instance later
757  // and run deserialization we run it on the proper type.
758  if (IsTypeSpecified(data)) {
759  fsData typeNameData = data.AsDictionary[Key_InstanceType];
760 
761  // we wrap everything in a do while false loop so we can break out it
762  do {
763  if (typeNameData.IsString == false) {
764  deserializeResult.AddMessage(Key_InstanceType + " value must be a string (in " + data + ")");
765  break;
766  }
767 
768  string typeName = typeNameData.AsString;
769  Type type = fsTypeCache.GetType(typeName);
770  if (type == null) {
771  deserializeResult += fsResult.Fail("Unable to locate specified type \"" + typeName + "\"");
772  break;
773  }
774 
775  if (storageType.IsAssignableFrom(type) == false) {
776  deserializeResult.AddMessage("Ignoring type specifier; a field/property of type " + storageType + " cannot hold an instance of " + type);
777  break;
778  }
779 
780  objectType = type;
781  } while (false);
782  }
783 
784  // We wait until here to actually Invoke_OnBeforeDeserialize because we do not
785  // have the correct set of processors to invoke until *after* we have resolved
786  // the proper type to use for deserialization.
787  processors = GetProcessors(objectType);
788 
789  if (deserializeResult.Failed)
790  return deserializeResult;
791 
792  Invoke_OnBeforeDeserialize(processors, storageType, ref data);
793 
794  // Construct an object instance if we don't have one already. We also need to construct
795  // an instance if the result type is of the wrong type, which may be the case when we
796  // have a versioned import graph.
797  if (ReferenceEquals(result, null) || result.GetType() != objectType) {
798  result = GetConverter(objectType, overrideConverterType).CreateInstance(data, objectType);
799  }
800 
801  // We call OnBeforeDeserializeAfterInstanceCreation here because we still want to invoke the
802  // method even if the user passed in an existing instance.
803  Invoke_OnBeforeDeserializeAfterInstanceCreation(processors, storageType, result, ref data);
804 
805  // NOTE: It is critically important that we pass the actual objectType down instead of
806  // using result.GetType() because it is not guaranteed that result.GetType()
807  // will equal objectType, especially because some converters are known to
808  // return dummy values for CreateInstance() (for example, the default behavior
809  // for structs is to just return the type of the struct).
810 
811  return deserializeResult += InternalDeserialize_4_Cycles(overrideConverterType, data, objectType, ref result);
812  }
813 
814  private fsResult InternalDeserialize_4_Cycles(Type overrideConverterType, fsData data, Type resultType, ref object result) {
815  if (IsObjectDefinition(data)) {
816  // NOTE: object references are handled at stage 1
817 
818  // If this is a definition, then we have a serialization invariant that this is the
819  // first time we have encountered the object (TODO: verify in the deserialization logic)
820 
821  // Since at this stage in the deserialization process we already have access to the
822  // object instance, so we just need to sync the object id to the references database
823  // so that when we encounter the instance we lookup this same object. We want to do
824  // this before actually deserializing the object because when deserializing the object
825  // there may be references to itself.
826 
827  int sourceId = int.Parse(data.AsDictionary[Key_ObjectDefinition].AsString);
828  _references.AddReferenceWithId(sourceId, result);
829  }
830 
831  // Nothing special, go through the standard deserialization logic.
832  return InternalDeserialize_5_Converter(overrideConverterType, data, resultType, ref result);
833  }

836  if (IsWrappedData(data)) {
837  data = data.AsDictionary[Key_Content];
838  }
839 
840  return GetConverter(resultType, overrideConverterType).TryDeserialize(data, ref result, resultType);
