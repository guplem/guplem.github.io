123  private static bool CanSerializeProperty(fsConfig config, PropertyInfo property, MemberInfo[] members, bool annotationFreeValue) {
124  // We don't serialize delegates
125  if (typeof(Delegate).IsAssignableFrom(property.PropertyType)) {
126  return false;
127  }
128 
129  var publicGetMethod = property.GetGetMethod(/*nonPublic:*/ false);
130  var publicSetMethod = property.GetSetMethod(/*nonPublic:*/ false);
131 
132  // We do not bother to serialize static fields.
133  if ((publicGetMethod != null && publicGetMethod.IsStatic) ||
134  (publicSetMethod != null && publicSetMethod.IsStatic)) {
135  return false;
136  }
137 
138  // Never serialize indexers. I can't think of a sane way to serialize/deserialize them, and they're normally wrappers around other fields anyway...
139  if (property.GetIndexParameters().Length > 0) {
140  return false;
141  }
142 
143  // If a property is annotated with one of the serializable attributes, then it should
144  // it should definitely be serialized.
145  //
146  // NOTE: We place this override check *after* the static check, because we *never*
147  // allow statics to be serialized.
148  if (config.SerializeAttributes.Any(t => fsPortableReflection.HasAttribute(property, t))) {
149  return true;
150  }
151 
152  // If the property cannot be both read and written to, we are not going to serialize it
153  // regardless of the default serialization mode
154  if (property.CanRead == false || property.CanWrite == false) {
155  return false;
156  }
157 
158  // Depending on the configuration options, check whether the property is automatic
159  // and if it has a public setter to determine whether it should be serialized
160  if ((publicGetMethod != null && (config.SerializeNonPublicSetProperties || publicSetMethod != null)) &&
161  (config.SerializeNonAutoProperties || IsAutoProperty(property, members))) {
162  return true;
163  }
164 
165 
166  // Otherwise, we don't bother with serialization
167  return annotationFreeValue;
