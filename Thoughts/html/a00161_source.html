17  // Internally we serialize an Object to the SceneAsset which only exists at editor time.
18  // Any time the object is serialized, we store the path provided by this Asset (assuming it was valid).
19  //
20  // This means that, come build time, the string path of the scene asset is always already stored, which if
21  // the scene was added to the build settings means it can be loaded.
22  //
23  // It is up to the user to ensure the scene exists in the build settings so it is loadable at runtime.
24  // To help with this, a custom PropertyDrawer displays the scene build settings state.
25  //
26  // Known issues:
27  // - When reverting back to a prefab which has the asset stored as null, Unity will show the property
28  // as modified despite having just reverted. This only happens on the fist time, and reverting again fix it.
29  // Under the hood the state is still always valid and serialized correctly regardless.

115  private void HandleBeforeSerialize()
116  {
117  // Asset is invalid but have Path to try and recover from
118  if (IsValidSceneAsset == false && string.IsNullOrEmpty(scenePath) == false)
119  {
120  sceneAsset = GetSceneAssetFromPath();
121  if (sceneAsset == null) scenePath = string.Empty;
122 
123  EditorSceneManager.MarkAllScenesDirty();
124  }
125  // Asset takes precendence and overwrites Path
126  else
127  {
128  scenePath = GetScenePathFromAsset();
129  }
130  }
131 
132  private void HandleAfterDeserialize()
133  {
134  EditorApplication.update -= HandleAfterDeserialize;
135  // Asset is valid, don't do anything - Path will always be set based on it when it matters
136  if (IsValidSceneAsset) return;
137 
138  // Asset is invalid but have path to try and recover from
139  if (string.IsNullOrEmpty(scenePath)) return;
140 
141  sceneAsset = GetSceneAssetFromPath();
142  // No asset found, path was invalid. Make sure we don't carry over the old invalid path
143  if (!sceneAsset) scenePath = string.Empty;
144 
145  if (!Application.isPlaying) EditorSceneManager.MarkAllScenesDirty();
146  }

378  private static bool QueryBuildSettingsStatus()
379  {
380  // If no version control provider, assume not readonly
381  if (!Provider.enabled) return false;
382 
383  // If we cannot checkout, then assume we are not readonly
384  if (!Provider.hasCheckoutSupport) return false;
385 
387  //if (UnityEditor.VersionControl.Provider.onlineState == UnityEditor.VersionControl.OnlineState.Offline)
388  // return true;
389 
390  // Try to get status for file
391  var status = Provider.Status("ProjectSettings/EditorBuildSettings.asset", false);
392  status.Wait();
393 
394  // If no status listed we can edit
395  if (status.assetList == null || status.assetList.Count != 1) return true;
396 
397  // If is checked out, we can edit
398  return !status.assetList[0].IsState(Asset.States.CheckedOutLocal);
399  }
